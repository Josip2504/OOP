fun main(args: Array<String>) {
    var args0 = args[0].toInt()
    var args1 = args[1].toInt()
    if (args0 % 2 == 0 && args1 % 2 == 0) {
        println("oba su parna")
    }else if (args0 % 2 != 0 && args1 % 2 != 0) {
        println("oba su neparna")
    }else {
        println("jedan je paran, drugi nije")
    }
}
